# Copy các lệnh này và chạy trong terminal:

# 1. Link repo local với GitHub (thay YOUR-USERNAME bằng username GitHub của bạn)
git remote add origin https://github.com/YOUR-USERNAME/the-smurf.git

# 2. Đổi branch sang main (nếu cần)
git branch -M main

# 3. Push code lên GitHub
git push -u origin main

# Nhập GitHub username và password (hoặc personal access token) khi được hỏi
